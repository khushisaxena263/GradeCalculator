     int eng=Integer.parseInt(t2.getText());
     int maths=Integer.parseInt(t3.getText());

     int phy=Integer.parseInt(t5.getText());
     int total=chem+maths+eng+phy;
     t6.setText(""+total);
     if(r1.isSelected()==true)
     {
         l1.setText("5 marks is deducted for dues");

     {
         l1.setText("no minus marking ");

    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {                                         
    try{
    float perc=Float.parseFloat(t7.getText());
    if(perc>=90)
    {
     t8.setText("A"); 
     l2.setText("PASS");
    } 
    else if(perc>=80)
    {
        t8.setText("B"); 
        l2.setText("PASS");
    }
    else if(perc>=60)
    {
        t8.setText("C");
        l2.setText("PASS");
    }
    else if(perc>=34)
    {
        t8.setText("D"); 
        l2.setText("PASS");
    }
    else
    {
        t8.setText("F");
        l2.setText("FAIL");
    }
    
    }catch(Exception e)
    {
         JOptionPane.showMessageDialog(null, "Input all the fields");
    }

    }
